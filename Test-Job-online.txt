function Resolve-TestJobLauncherBatPath {
    if ($env:TESTJOB_SELF_BAT -and (Test-Path -LiteralPath $env:TESTJOB_SELF_BAT)) {
        return $env:TESTJOB_SELF_BAT
    }

    $candidates = @()
    if ($Script:OutputDir) {
        $candidates += (Join-Path $Script:OutputDir "Test-Job.bat")
    }
    $candidates += (Join-Path (Get-Location).Path "Test-Job.bat")

    foreach ($candidate in ($candidates | Select-Object -Unique)) {
        if ($candidate -and (Test-Path -LiteralPath $candidate)) {
            return $candidate
        }
    }

    try {
        $selfPid = $PID
        $seen = New-Object System.Collections.Generic.HashSet[int]
        while ($selfPid -and -not $seen.Contains($selfPid)) {
            [void]$seen.Add($selfPid)
            $proc = Get-CimInstance Win32_Process -Filter "ProcessId = $selfPid" -ErrorAction Stop
            if (-not $proc) { break }
            $cmd = [string]$proc.CommandLine
            if ($cmd -match '([A-Za-z]:\\[^"]*Test-Job\.bat)') {
                $batPath = $Matches[1]
                if (Test-Path -LiteralPath $batPath) { return $batPath }
            }
            $selfPid = [int]$proc.ParentProcessId
        }
    } catch {}

    return ""
}

function Get-TestJobBatUpdateState {
    if ($Script:BatUpdateState) { return $Script:BatUpdateState }
    $state = [pscustomobject]@{
        UpdateAvailable = ($env:TESTJOB_BAT_UPDATE_AVAILABLE -eq "1")
        SelfBatPath = (Resolve-TestJobLauncherBatPath)
        CacheDir = if ($env:TESTJOB_CACHE_DIR) { $env:TESTJOB_CACHE_DIR } else { (Join-Path ([System.IO.Path]::GetTempPath()) "TestJob") }
        LegacyBat = $false
    }
    if ($state.UpdateAvailable) {
        $Script:BatUpdateState = $state
        return $state
    }
    if ([string]::IsNullOrWhiteSpace($state.SelfBatPath) -or -not (Test-Path -LiteralPath $state.SelfBatPath)) {
        $Script:BatUpdateState = $state
        return $state
    }
    try {
        $localVersion = "0"
        $versionLine = Select-String -LiteralPath $state.SelfBatPath -Pattern 'TESTJOB_BAT_VERSION=' -SimpleMatch -ErrorAction Stop | Select-Object -First 1
        if ($versionLine) {
            $localVersion = (($versionLine.Line -split 'TESTJOB_BAT_VERSION=')[1]).Trim('"')
        } else {
            $state.LegacyBat = $true
            $state.UpdateAvailable = $true
        }
        $remoteVersion = ""
        try {
            $remoteVersion = ((Invoke-WebRequest -UseBasicParsing -Uri $Script:DefaultRemoteBatVersionUrl -ErrorAction Stop).Content | Out-String).Trim()
        } catch {
            if (-not $state.LegacyBat) {
                throw
            }
        }
        if (-not $state.LegacyBat) {
            if ([string]::IsNullOrWhiteSpace($remoteVersion) -or $remoteVersion -le $localVersion) {
                $Script:BatUpdateState = $state
                return $state
            }
        }
        if (-not (Test-Path -LiteralPath $state.CacheDir)) {
            [void](New-Item -ItemType Directory -Path $state.CacheDir -Force)
        }
        $nextBat = Join-Path $state.CacheDir "Test-Job-next.bat"
        $nextVer = Join-Path $state.CacheDir "Test-Job-next.version.txt"
        $nextFlag = Join-Path $state.CacheDir "Test-Job-next.ready"
        $nextHelper = Join-Path $state.CacheDir "Test-Job-next-update.cmd"
        $batContent = [string](Invoke-WebRequest -UseBasicParsing -Uri $Script:DefaultRemoteBatUrl -ErrorAction Stop).Content
        if ($batContent -notmatch 'TESTJOB_BAT_VERSION=' -or $batContent -notmatch '::PowerShell_Start') {
            $Script:BatUpdateState = $state
            return $state
        }
        if (-not [string]::IsNullOrWhiteSpace($remoteVersion) -and $batContent -notmatch ('TESTJOB_BAT_VERSION=' + [regex]::Escape($remoteVersion))) {
            $Script:BatUpdateState = $state
            return $state
        }
        Set-Content -LiteralPath $nextBat -Value $batContent -Encoding ASCII
        Set-Content -LiteralPath $nextVer -Value $remoteVersion -Encoding ASCII
        Set-Content -LiteralPath $nextFlag -Value "ready" -Encoding ASCII
        $state.UpdateAvailable = $true
        $state.NextBatPath = $nextBat
        $state.NextVersionPath = $nextVer
        $state.NextFlagPath = $nextFlag
        $state.HelperPath = $nextHelper
    } catch {}
    $Script:BatUpdateState = $state
    return $state
}

function Start-TestJobBatReplacement {
    $state = Get-TestJobBatUpdateState
    try {
        if (-not $state.UpdateAvailable) { return }
        if ([string]::IsNullOrWhiteSpace($state.SelfBatPath) -or -not (Test-Path -LiteralPath $state.SelfBatPath)) { return }
        if ([string]::IsNullOrWhiteSpace($state.NextFlagPath) -or
            [string]::IsNullOrWhiteSpace($state.NextBatPath) -or
            [string]::IsNullOrWhiteSpace($state.HelperPath)) { return }
        if (-not (Test-Path -LiteralPath $state.NextFlagPath) -or -not (Test-Path -LiteralPath $state.NextBatPath)) { return }
        @"
@echo off
setlocal EnableExtensions
set "SRC=$($state.NextBatPath)"
set "DST=$($state.SelfBatPath)"
set "VER=$($state.NextVersionPath)"
set "FLAG=$($state.NextFlagPath)"
set "SELF=$($state.HelperPath)"
set /a TRYCOUNT=0
:retry
ping 127.0.0.1 -n 3 >nul
copy /y "%SRC%" "%DST%" >nul 2>nul
if not errorlevel 1 goto copied
set /a TRYCOUNT+=1
if %TRYCOUNT% LSS 20 goto retry
goto end
:copied
del /q "%SRC%" "%VER%" "%FLAG%" "%SELF%" >nul 2>nul
:end
"@ | Set-Content -LiteralPath $state.HelperPath -Encoding ASCII
        Start-Process -FilePath "cmd.exe" -ArgumentList "/c","`"$($state.HelperPath)`"" -WindowStyle Hidden | Out-Null
    } catch {}
}

function Set-TextBoxCueBanner {
    param(
        [System.Windows.Forms.TextBox]$TextBox,
        [string]$Text
    )
    if (-not $TextBox -or [string]::IsNullOrWhiteSpace($Text)) { return }
    if (-not ("TestJobNativeMethods" -as [type])) {
        Add-Type -TypeDefinition @"
using System;
using System.Runtime.InteropServices;
public static class TestJobNativeMethods {
    [DllImport("user32.dll", CharSet = CharSet.Unicode)]
    public static extern IntPtr SendMessage(IntPtr hWnd, int msg, IntPtr wParam, string lParam);
}
"@
    }
    $EM_SETCUEBANNER = 0x1501
    [void][TestJobNativeMethods]::SendMessage($TextBox.Handle, $EM_SETCUEBANNER, [IntPtr]::Zero, $Text)
}

function Set-TestJobProgress {
    param(
        [int]$Percent,
        [string]$Text
    )
    if ($env:TESTJOB_WORKER -eq "1" -and $env:TESTJOB_PROGRESS_PATH) {
        try {
            $entry = [pscustomobject]@{
                Time = (Get-Date).ToString("HH:mm:ss")
                Percent = [Math]::Max(0, [Math]::Min(100, $Percent))
                Text = $Text
            }
            Add-Content -LiteralPath $env:TESTJOB_PROGRESS_PATH -Value ($entry | ConvertTo-Json -Compress) -Encoding UTF8
        } catch {}
        return
    }
    if (-not $Script:ProgressForm) { return }
    $update = {
        param([int]$p, [string]$t)
        try {
            $value = [math]::Max(0, [math]::Min(100, $p))
            $Script:ProgressBar.Value = $value
            $Script:ProgressStatus.Text = $t
            if ($Script:ProgressPercent) { $Script:ProgressPercent.Text = "$value%" }
            if ($Script:ProgressPhases) {
                for ($i = 0; $i -lt $Script:ProgressPhases.Count; $i++) {
                    $phaseBoundary = @(0, 20, 38, 55, 72, 86)[$i]
                    if ($value -ge $phaseBoundary) {
                        $Script:ProgressPhases[$i].ForeColor = [System.Drawing.Color]::FromArgb(63, 185, 80)
                    } else {
                        $Script:ProgressPhases[$i].ForeColor = [System.Drawing.Color]::FromArgb(104, 116, 130)
                    }
                }
                if ($value -ge 42 -and $value -lt 58) { $Script:ProgressPhases[2].ForeColor = [System.Drawing.Color]::FromArgb(88, 166, 255) }
                elseif ($value -ge 58 -and $value -lt 70) { $Script:ProgressPhases[3].ForeColor = [System.Drawing.Color]::FromArgb(88, 166, 255) }
                elseif ($value -ge 70 -and $value -lt 86) { $Script:ProgressPhases[4].ForeColor = [System.Drawing.Color]::FromArgb(88, 166, 255) }
                elseif ($value -ge 86 -and $value -lt 100) { $Script:ProgressPhases[5].ForeColor = [System.Drawing.Color]::FromArgb(88, 166, 255) }
            }
            if ($Script:ProgressLog) {
                $line = "{0:HH:mm:ss}  {1}" -f (Get-Date), $t
                $Script:ProgressLog.AppendText($line + [Environment]::NewLine)
                $Script:ProgressLog.SelectionStart = $Script:ProgressLog.TextLength
                $Script:ProgressLog.ScrollToCaret()
            }
        } catch {}
    }
    try {
        if ($Script:ProgressForm.InvokeRequired) {
            [void]$Script:ProgressForm.BeginInvoke($update, @($Percent, $Text))
        } else {
            & $update $Percent $Text
        }
    } catch {}
}

    param([string]$Text = "Gereed")
    if (-not $Script:ProgressForm) { return }

function HtmlEncode {
    param([object]$Value)
    if ($null -eq $Value) { return "" }
    return [System.Net.WebUtility]::HtmlEncode([string]$Value)
}

function AttrEncode {
    param([object]$Value)
    return (HtmlEncode $Value).Replace("'", "&#39;")
}

function Format-ShortDate {
    param([object]$Value)
    if (-not $Value) { return "" }
    try {
        $dt = [datetime]$Value
        return $dt.ToString("dd MMM yyyy")
    } catch {
        return [string]$Value
    }
}

function Format-FilePart {
    param([string]$Value)
    if ([string]::IsNullOrWhiteSpace($Value)) { return "" }
    $clean = (($Value.Trim()) -replace '[\\/:*?"<>|\s]+', '-')
    $clean = ($clean -replace '-{2,}', '-').Trim('-')
    return $clean
}

function Limit-FilePart {
    param(
        [string]$Value,
        [int]$MaxLength = 40
    )
    $clean = Format-FilePart $Value
    if (-not $clean) { return "" }
    if ($clean.Length -le $MaxLength) { return $clean }
    return $clean.Substring(0, $MaxLength).Trim('-')
}

function Format-TicketNumber {
    param([string]$Value)
    $raw = [string]$Value
    if ([string]::IsNullOrWhiteSpace($raw)) { return "" }
    $cleaned = $raw.Trim().ToUpperInvariant()
    if ($cleaned.StartsWith("VM")) {
        $cleaned = $cleaned.Substring(2).Trim()
    }
    $digits = -join ($cleaned.ToCharArray() | Where-Object { [char]::IsDigit($_) })
    if ($digits.Length -ne 8) { return "" }
    return "VM{0}-{1}" -f $digits.Substring(0, 4), $digits.Substring(4, 4)
}

function Get-TicketUrl {
    param([string]$Ticket)
    if (-not $Ticket) { return "" }
    return "https://vertimart.topdesk.net/tas/secure/incident?action=lookup&lookup=naam&lookupValue=$Ticket"
}

function Get-ReportFileName {
    param([object]$InputData)
    $parts = @()
    $ticket = Limit-FilePart (Format-TicketNumber $InputData.Ticketnummer) 20
    $praktijk = Limit-FilePart $InputData.Praktijk 40
    $naam = Limit-FilePart $InputData.Naam 40
    $licentie = Limit-FilePart $InputData.Licentie 28
    foreach ($part in @($ticket, $praktijk, $naam, $licentie)) {
        if ($part) { $parts += $part }
    }
    $prefix = if ($parts.Count) { ($parts -join "_") + "_" } else { "" }
    $fileName = "${prefix}Test-Job_Rapport.html"
    if ($fileName.Length -le 150) { return $fileName }
    return ($fileName.Substring(0, 150).TrimEnd('_', '-')) + ".html"
}

function Get-WorstClass {
    param([string[]]$Classes)
    $rank = @{ muted = 0; ok = 1; warn = 2; bad = 3 }
    $winner = "muted"
    foreach ($cls in $Classes) {
        $name = if ($cls) { $cls } else { "muted" }
        if ($rank[$name] -gt $rank[$winner]) { $winner = $name }
    }
    return $winner

function Invoke-ParallelTaskScripts {
    param(
        [object[]]$Tasks,
        [int]$MaxWorkers = 6
    )

    $results = @{}
    if (-not $Tasks -or $Tasks.Count -eq 0) { return $results }

    $pool = [System.Management.Automation.Runspaces.RunspaceFactory]::CreateRunspacePool(1, $MaxWorkers)
    $pool.Open()
    $jobs = @()
    try {
        foreach ($task in $Tasks) {
            $ps = [powershell]::Create()
            $null = $ps.AddScript($task.ScriptText)
            $taskArgs = @()
            if ($null -ne $task.Arguments) {
                $taskArgs = @($task.Arguments)
            }
            foreach ($arg in $taskArgs) {
                $null = $ps.AddArgument($arg)
            }
            $ps.RunspacePool = $pool
            $jobs += [pscustomobject]@{
                Name = $task.Name
                Pipe = $ps
                Result = $ps.BeginInvoke()
                Default = $task.Default
            }
        }

        foreach ($job in $jobs) {
            try {
                $output = $job.Pipe.EndInvoke($job.Result)
                $results[$job.Name] = @($output)
            } catch {
                [void]$Script:CollectionErrors.Add([pscustomobject]@{
                    Source = $job.Name
                    Error = $_.Exception.Message
                    Type = $_.Exception.GetType().Name
                })
                $results[$job.Name] = $job.Default
            } finally {
                $job.Pipe.Dispose()
            }
        }
    } finally {
        $pool.Close()
        $pool.Dispose()
    }
    return $results
}

function Get-CpuInfo {
    $cpu = Get-CimInstance Win32_Processor | Select-Object -First 1
    $name = ($cpu.Name -replace "\s+", " ").Trim()
    $generation = 0
    $year = $null
    if ($name -match "Intel.*Ultra") {
        $generation = 15
        $year = 2023
    } elseif ($name -match "i\d-(\d{4,5})") {
        $model = $Matches[1]
        if ($model.Length -eq 4) { $generation = [int]$model.Substring(0, 1) } else { $generation = [int]$model.Substring(0, 2) }
        $yearMap = @{ 15 = 2025; 14 = 2023; 13 = 2022; 12 = 2021; 11 = 2020; 10 = 2019; 9 = 2018; 8 = 2017 }
        if ($yearMap.ContainsKey($generation)) { $year = $yearMap[$generation] }
    } elseif ($name -match "Ryzen\s+\d\s+(\d{4})") {
        $generation = [int]$Matches[1].Substring(0, 1)
        $amdYears = @{ 7 = 2022; 5 = 2020; 3 = 2019 }
        if ($amdYears.ContainsKey($generation)) { $year = $amdYears[$generation] }
    }
    [pscustomobject]@{
        Name = $name
        Cores = [int]$cpu.NumberOfCores
        Logical = [int]$cpu.NumberOfLogicalProcessors
        Generation = $generation
        LaunchYear = $year
    }
}

        $client.Dispose()

function Test-DomainSet {
    param(
        [string[]]$DomainsExtensive = $Script:DomainsExtensive,
        [string[]]$DomainsBasic = $Script:DomainsBasic,
        [string[]]$TrustedIssuers = $Script:TrustedIssuers
    )
    $all = @()
    foreach ($d in $DomainsExtensive) { $all += [pscustomobject]@{ Domain = $d; Extensive = $true } }
    foreach ($d in $DomainsBasic) { $all += [pscustomobject]@{ Domain = $d; Extensive = $false } }

    $worker = {
        param($Entry, $TrustedIssuerText)
        $TrustedIssuers = @($TrustedIssuerText -split "\|")

        function Get-HostFromEndpointLocal {
            param([string]$Endpoint)
            return (($Endpoint -split "/")[0]).Trim()
        }

        function Test-DnsEndpointLocal {

        }

        function Test-SslEndpointLocal {
            param([string]$Endpoint)
            $hostName = Get-HostFromEndpointLocal $Endpoint
            $client = [System.Net.Sockets.TcpClient]::new()
            $stream = $null
            try {
                $connectTask = $client.ConnectAsync($hostName, 443)
                if (-not $connectTask.Wait(5000)) {
                    return [pscustomobject]@{ Issuer = "Fout: timeout"; Tls = "Mislukt" }
                }
                $stream = [System.Net.Security.SslStream]::new($client.GetStream(), $false, ({ $true } -as [System.Net.Security.RemoteCertificateValidationCallback]))
                $stream.AuthenticateAsClient($hostName)
                $cert = [System.Security.Cryptography.X509Certificates.X509Certificate2]::new($stream.RemoteCertificate)
                $issuer = $cert.Issuer
                $org = if ($issuer -match "O=([^,]+)") { $Matches[1] } else { $issuer }
                return [pscustomobject]@{ Issuer = $org; Tls = $stream.SslProtocol.ToString() }
            } catch {
                return [pscustomobject]@{ Issuer = "Fout: $($_.Exception.Message)"; Tls = "Mislukt" }
            } finally {

            }
        }

        function Test-HttpEndpointLocal {
            param([string]$Endpoint)
            $url = if ($Endpoint -match "^https?://") { $Endpoint } else { "https://$Endpoint" }
            $sw = [System.Diagnostics.Stopwatch]::StartNew()
            try {
                $r = Invoke-WebRequest -Uri $url -UseBasicParsing -TimeoutSec 10 -ErrorAction Stop
                $sw.Stop()
                return [pscustomobject]@{ Status = "OK ($($r.StatusCode))"; Code = $r.StatusCode; TimeMs = [math]::Round($sw.Elapsed.TotalMilliseconds) }
            } catch {
                $sw.Stop()
                if ($_.Exception.Response -and $_.Exception.Response.StatusCode) {
                    $code = [int]$_.Exception.Response.StatusCode
                    return [pscustomobject]@{ Status = "OK ($code)"; Code = $code; TimeMs = [math]::Round($sw.Elapsed.TotalMilliseconds) }
                }
                return [pscustomobject]@{ Status = "Mislukt"; Code = 0; TimeMs = -1 }
            }
        }

        $tcp = Test-TcpEndpointLocal $Entry.Domain
        $ssl = Test-SslEndpointLocal $Entry.Domain
        $issuer = $ssl.Issuer
        $sslStatus = "Niet vertrouwd"
        if ($issuer -like "Fout:*") {
            $sslStatus = "FOUT"
        } else {
            foreach ($trusted in $TrustedIssuers) {
                if ($issuer -like "*$trusted*") { $sslStatus = "OK"; break }
            }
        }
        $http = if ($Entry.Extensive -and $tcp.Open) {
            Test-HttpEndpointLocal $Entry.Domain
        } else {
            [pscustomobject]@{ Status = ""; Code = 0; TimeMs = -1 }
        }

        [pscustomobject]@{
            Domain = [string]$Entry.Domain
            Extensive = [bool]$Entry.Extensive
            Dns = [string](Test-DnsEndpointLocal $Entry.Domain)
            Port443 = [string]$tcp.Label
            TcpLatencyMs = [int]$tcp.Latency
            TcpLatency = if ($tcp.Latency -ge 0) { "$($tcp.Latency)ms" } else { "timeout" }
            SslIssuer = [string]$issuer
            SslStatus = [string]$sslStatus
            Tls12 = if ($ssl.Tls -match "Tls12|Tls13") { "OK" } else { [string]$ssl.Tls }
            Http = [string]$http.Status
            HttpTimeMs = [int]$http.TimeMs
            HttpTime = if ($http.TimeMs -ge 0) { "$($http.TimeMs)ms" } else { "-" }
        }
    }

    $jobs = @()
    try {
        foreach ($entry in $all) {
            $jobs += Start-Job -ScriptBlock $worker -ArgumentList $entry, ($TrustedIssuers -join "|")
        }

        Wait-Job -Job $jobs -Timeout 35 | Out-Null
        $results = @($jobs | Receive-Job -ErrorAction SilentlyContinue)
    } finally {
        $jobs | Remove-Job -Force -ErrorAction SilentlyContinue
    }

    $order = @{}
    for ($i = 0; $i -lt $all.Count; $i++) { $order[$all[$i].Domain] = $i }
    $normalizedResults = @($results | Where-Object { $_ -and $_.Domain } | Sort-Object { $order[$_.Domain] })

    [pscustomobject]@{
        Extensive = @($normalizedResults | Where-Object { $_.Extensive })
        Basic = @($normalizedResults | Where-Object { -not $_.Extensive })
        Failures = @($normalizedResults | Where-Object { $_.SslStatus -eq "FOUT" -or $_.Dns -eq "Mislukt" })
    }
}

function Get-InstalledApps {
    $paths = @(
        "HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall\*",
        "HKLM:\Software\Wow6432Node\Microsoft\Windows\CurrentVersion\Uninstall\*"
    )
    foreach ($path in $paths) {
        Get-ItemProperty $path -ErrorAction SilentlyContinue |
            Where-Object {
                $_.DisplayName -and $_.SystemComponent -ne 1 -and
                $_.DisplayName -notmatch "^(KB\d+|Update for|Cumulatieve update|Servicing Stack|Dynamic Update for)"
            } |
            ForEach-Object {
                [pscustomobject]@{
                    Name = $_.DisplayName
                    Version = $_.DisplayVersion
                    Publisher = $_.Publisher
                    InstallDate = $_.InstallDate
                }
            }
    }
}

function Split-InstalledApps {
    param([object[]]$Apps)
    $updatePatterns = @(
        "update voor", "Update for", "Security Update", "beveiligingsinformatie",
        "antimalwareplatform", "\(KB\d+\)", "^KB\d+:", "Microsoft.VCLibs",
        "Microsoft.NET.Native", "Microsoft.Windows", "Update for Microsoft",
        "Cumulatieve update", "Dynamic Update for", "Servicing Stack"
    )
    $software = @()
    $updates = @()
    foreach ($app in $Apps) {
        $isUpdate = $false
        foreach ($pattern in $updatePatterns) {
            if ($app.Name -match $pattern) { $isUpdate = $true; break }
        }
        if ($app.Name -like "{*") { $isUpdate = $true }
        if ($isUpdate) { $updates += $app } else { $software += $app }
    }
    [pscustomobject]@{
        Software = @($software | Sort-Object Name)
        Updates = @($updates | Sort-Object Name)
    }
}

function Get-AppChecks {
    param([object[]]$Apps)
    $checks = New-Object System.Collections.Generic.List[object]

    $officeWarnings = @()
    foreach ($app in $Apps) {
        if ($app.Name -match "Microsoft Office (Professional|Standard|Home|Small Business)\s*(20\d{2})" -and [int]$Matches[2] -lt 2021) {
            $officeWarnings += "Office $($Matches[2])"
        }
        if ($app.Name -match "Microsoft (Word|Excel|Outlook|PowerPoint|Access|Publisher)" -and $app.Version -match "(20\d{2})" -and [int]$Matches[1] -lt 2021) {
            $officeWarnings += "$($Matches[1])"
        }
        if ($app.Name -match "Microsoft 365 Apps" -and $app.Version -match "(20\d{2})" -and [int]$Matches[1] -lt 2021) {
            $officeWarnings += "Microsoft 365 Apps (verouderd)"
        }
    }
    if ($officeWarnings) {
        [void]$checks.Add([pscustomobject]@{ Category = "Office"; Status = "warn"; Message = (($officeWarnings | Select-Object -Unique) -join "; ") + " - upgrade naar Office 2021 of Microsoft 365" })
    } else {
        [void]$checks.Add([pscustomobject]@{ Category = "Office"; Status = "ok"; Message = "Office is up-to-date" })
    }

    $consumerAvPatterns = @("mcafee", "norton", "avast", "avg", "avira", "kaspersky", "panda dome", "webroot secureanywhere", "totalav", "f-secure safe")
    $businessMarkers = @("microsoft defender", "crowdstrike falcon", "endpoint", "business", "gravityzone", "intercept x", "enterprise", "corporate", "for business")
    $consumerFound = @()
    foreach ($app in $Apps) {
        $low = ([string]$app.Name).ToLowerInvariant()
        if ($businessMarkers | Where-Object { $low.Contains($_) }) { continue }
        if (($consumerAvPatterns | Where-Object { $low.Contains($_) }) -or $low -match "bitdefender.*\b(total|internet|antivirus)\b|malwarebytes.*\bpremium\b|trend micro.*\b(maximum|internet security)\b") {
            $consumerFound += $app.Name
        }
    }
    if ($consumerFound) {
        [void]$checks.Add([pscustomobject]@{ Category = "Consumenten virusscanner"; Status = "warn"; Message = "$($consumerFound[0]) gedetecteerd - kan Exquise, netwerkverkeer of performance beinvloeden" })
    } else {
        [void]$checks.Add([pscustomobject]@{ Category = "Consumenten virusscanner"; Status = "ok"; Message = "Niet gedetecteerd" })
    }

    $javaWarnings = @()
    foreach ($app in $Apps) {
        if ($app.Name -match "java" -and ([string]$app.Version).Trim() -match "^(1\.|8\.|7\.|6\.)") {
            $javaWarnings += "$($app.Name) ($($app.Version))"
        }
    }
    if ($javaWarnings) {
        [void]$checks.Add([pscustomobject]@{ Category = "Java"; Status = "warn"; Message = ($javaWarnings -join "; ") + " - upgrade naar Java 11 of nieuwer" })
    } else {
        [void]$checks.Add([pscustomobject]@{ Category = "Java"; Status = "ok"; Message = "Niet gedetecteerd of up-to-date" })
    }

    $ace = @()
    foreach ($app in $Apps) {
        if ($app.Name -match "access" -and $app.Name -match "engine" -and $app.Version -match "(20\d{2})" -and [int]$Matches[1] -le 2016) {
            $ace += "$($app.Name) ($($app.Version))"
        }
    }
    $aceRegistry = Test-Path "HKLM:\SOFTWARE\Microsoft\Office\ClickToRun\Registry\Machine\Software\Wow6432Node\Microsoft\Office\16.0\Access Connectivity Engine\Engines"
    if ($ace -or $aceRegistry) {
        $msg = @()
        if ($ace) { $msg += (($ace -join "; ") + " (EOL, <= 2016)") }
        if ($aceRegistry) { $msg += "Click-to-Run registry key aanwezig" }
        [void]$checks.Add([pscustomobject]@{ Category = "Access Database Engine"; Status = "warn"; Message = ($msg -join " | ") })
    } else {
        [void]$checks.Add([pscustomobject]@{ Category = "Access Database Engine"; Status = "ok"; Message = "Niet gedetecteerd" })
    }

    $fp = @()
    foreach ($app in $Apps) {
        if ($app.Name -match "florida probe|fp32") {
            $detail = $app.Name
            if ($app.Version) { $detail += " v$($app.Version)" }
            if ($app.InstallDate) { $detail += " ($($app.InstallDate))" }
            $fp += $detail
        }
    }
    if ($fp) {
        [void]$checks.Add([pscustomobject]@{ Category = "Florida Probe"; Status = "warn"; Message = ($fp -join "; ") })
    } else {
        [void]$checks.Add([pscustomobject]@{ Category = "Florida Probe"; Status = "ok"; Message = "Niet gedetecteerd" })
    }

    $polyStudio = @()
    foreach ($app in $Apps) {
        $name = [string]$app.Name
        if ($name -match "^HP\s+Poly\s+Studio\s+App$" -or $name -match "^HP\s+Poly\s+Studio\s+Desktop\s+App$") {
            $polyStudio += if ($app.Version) { "$name ($($app.Version))" } else { $name }
        }
    }
    if ($polyStudio) {
        [void]$checks.Add([pscustomobject]@{ Category = "Poly Studio app"; Status = "warn"; Message = ($polyStudio | Select-Object -Unique) -join "; " })
    } else {
        [void]$checks.Add([pscustomobject]@{ Category = "Poly Studio app"; Status = "ok"; Message = "Niet gedetecteerd" })
    }

    return $checks
}

function Get-WindowsInstallDate {
    param([object[]]$Apps)
    $installDate = Get-RegValue "HKLM:\SOFTWARE\Microsoft\Windows NT\CurrentVersion" "InstallDate" $null
    if ($installDate) {
        return [pscustomobject]@{ Date = ([DateTimeOffset]::FromUnixTimeSeconds([int64]$installDate).DateTime).ToString("yyyy-MM-dd"); Source = "Windows InstallDate" }
    }
    $profile = Get-ChildItem "C:\Users" -Directory -ErrorAction SilentlyContinue |
        Where-Object { $_.Name -notin @("Public", "Default", "Default User", "All Users") } |
        Sort-Object CreationTime |
        Select-Object -First 1
    if ($profile) { return [pscustomobject]@{ Date = $profile.CreationTime.ToString("yyyy-MM-dd"); Source = "eerste gebruikersprofiel" } }
    $candidate = $Apps | Where-Object { $_.InstallDate -match "^\d{8}$" } | Sort-Object InstallDate | Select-Object -First 1
    if ($candidate) {
        $s = [string]$candidate.InstallDate
        return [pscustomobject]@{ Date = "$($s.Substring(0,4))-$($s.Substring(4,2))-$($s.Substring(6,2))"; Source = "oudste app-installatie" }
    }
    [pscustomobject]@{ Date = ""; Source = "onbekend" }
}

function Get-ExquiseLogs {
    $cutoff = (Get-Date).AddDays(-7)
    $maxSize = 5MB
    $sources = New-Object System.Collections.Generic.List[object]

    function Add-LogFiles {
        param(
            [string]$SourceName,
            [string]$Path,
            [string]$Filter,
            [switch]$Recurse
        )
        if ([string]::IsNullOrWhiteSpace($Path)) { return }
        $files = Get-ChildItem -Path $Path -Filter $Filter -File -Recurse:$Recurse -ErrorAction SilentlyContinue |
            Where-Object {
                $_.LastWriteTime -ge $cutoff -and $_.Length -le $maxSize
            }
        foreach ($file in $files) {
            $content = ""
            try { $content = Get-Content -Path $file.FullName -Raw -Encoding UTF8 -ErrorAction Stop } catch { $content = "[Kon logbestand niet lezen: $($_.Exception.Message)]" }
            [void]$sources.Add([pscustomobject]@{
                Source = $SourceName
                FileName = $file.Name
                Modified = $file.LastWriteTime
                Path = $file.FullName
                Content = $content
            })
        }
    }

    if ($env:LOCALAPPDATA) {
        Add-LogFiles "Exquise ClickOnce" (Join-Path $env:LOCALAPPDATA "Apps\2.0") "FEExquise*.log" -Recurse
        Add-LogFiles "IAS log" (Join-Path $env:LOCALAPPDATA "Apps\2.0") "ias.log" -Recurse
        $msixPackages = Get-ChildItem -Path (Join-Path $env:LOCALAPPDATA "Packages\Vertimart.Exquise_*") -Directory -ErrorAction SilentlyContinue
        foreach ($package in $msixPackages) {
            Add-LogFiles "Exquise MSIX" (Join-Path $package.FullName "LocalCache\Local\Vertimart\Exquise\logs") "FEExquise*.log"
        }
    }
    if ($env:APPDATA) {
        $repair = Join-Path $env:APPDATA "Exquise\Exquise RepairTool\Logs\RepairToolLog.txt"
        if (Test-Path $repair) {
            $file = Get-Item $repair
            if ($file.LastWriteTime -ge $cutoff -and $file.Length -le $maxSize) {
                $content = Get-Content -Path $file.FullName -Raw -Encoding UTF8 -ErrorAction SilentlyContinue
                [void]$sources.Add([pscustomobject]@{
                    Source = "RepairToolLog"
                    FileName = $file.Name
                    Modified = $file.LastWriteTime
                    Path = $file.FullName
                    Content = $content
                })
            }
        }
    }
    return $sources
}

function Get-LanScan {
    param(
        [string[]]$SonosOuis = $Script:SonosOuis,
        [string[]]$YealinkOuis = $Script:YealinkOuis
    )
    $default = [pscustomobject]@{ Subnet = ""; Total = 0; SonosCount = 0; SonosFound = $false; YealinkCount = 0; YealinkFound = $false; Devices = @() }
    $ipConfig = Get-NetIPConfiguration -ErrorAction SilentlyContinue | Where-Object { $_.IPv4DefaultGateway -and $_.IPv4Address } | Select-Object -First 1
    if (-not $ipConfig) { return $default }
    $ip = ($ipConfig.IPv4Address | Select-Object -First 1).IPAddress
    $parts = $ip.Split(".")
    if ($parts.Count -lt 4) { return $default }
    $networkBase = ($parts[0..2] -join ".")

    $pool = [System.Management.Automation.Runspaces.RunspaceFactory]::CreateRunspacePool(1, 120)
    $pool.Open()
    $jobs = @()
    $scriptBlock = {
        param($Address)
        try {
            $ping = [System.Net.NetworkInformation.Ping]::new()
            $reply = $ping.Send($Address, 750)
            if ($reply.Status -eq "Success") { return "$Address|$($reply.RoundtripTime)" }
        } catch {}
        return $null
    }
    try {
        foreach ($i in 1..254) {
            $ps = [powershell]::Create().AddScript($scriptBlock).AddArgument("$networkBase.$i")
            $ps.RunspacePool = $pool
            $jobs += [pscustomobject]@{ Pipe = $ps; Result = $ps.BeginInvoke() }
        }
        while (($jobs | Where-Object { -not $_.Result.IsCompleted }).Count -gt 0) {
            Start-Sleep -Milliseconds 100
        }
        $found = foreach ($job in $jobs) {
            $r = $job.Pipe.EndInvoke($job.Result)
            $job.Pipe.Dispose()
            if ($r) { $r }
        }
    } finally {
        $pool.Close()
        $pool.Dispose()
    }

    $neighbors = Get-NetNeighbor -AddressFamily IPv4 -ErrorAction SilentlyContinue
    $devices = foreach ($entry in $found) {
        $p = $entry -split "\|"
        $address = $p[0]
        $latency = if ($p.Count -gt 1) { $p[1] } else { "" }
        $neighbor = $neighbors | Where-Object { $_.IPAddress -eq $address -and $_.State -ne "Incomplete" } | Select-Object -First 1
        $mac = if ($neighbor) { $neighbor.LinkLayerAddress.ToUpper().Replace("-", ":") } else { "Niet gevonden" }
        $marker = ""
        if ($mac -ne "Niet gevonden" -and $mac.Length -ge 8) {
            $oui = $mac.Substring(0, 8)
            if ($SonosOuis -contains $oui) {
                $marker = "*"
            }
            elseif ($YealinkOuis -contains $oui) {
                $marker = "^"
            }
        }
        $hostName = ""
        try {
            $async = [System.Net.Dns]::BeginGetHostEntry($address, $null, $null)
            if ($async.AsyncWaitHandle.WaitOne(120)) {
                $dns = [System.Net.Dns]::EndGetHostEntry($async)
                if ($dns -and $dns.HostName) { $hostName = $dns.HostName }
            }
            $async.AsyncWaitHandle.Close()
        } catch {}
        [pscustomobject]@{
            Marker = $marker
            Ip = $address
            Mac = $mac
            HostName = $hostName
            LatencyMs = $latency
        }
    }
    $ordered = @($devices | Sort-Object { [version]$_.Ip })

    $sonos = @($ordered | Where-Object { $_.Marker -eq "*" })
    $yealink = @($ordered | Where-Object { $_.Marker -eq "^" })
    [pscustomobject]@{
        Subnet = $networkBase
        Total = $ordered.Count
        SonosCount = $sonos.Count
        SonosFound = $sonos.Count -gt 0
        YealinkCount = $yealink.Count
        YealinkFound = $yealink.Count -gt 0
        Devices = $ordered
    }
}

function Copy-FileToClipboard {
    param([string]$Path)
    try {
        Add-Type -AssemblyName System.Windows.Forms
        $collection = [System.Collections.Specialized.StringCollection]::new()
        [void]$collection.Add((Resolve-Path $Path).Path)
        $data = [System.Windows.Forms.DataObject]::new()
        $data.SetFileDropList($collection)
        $data.SetText((Resolve-Path $Path).Path)
        for ($i = 0; $i -lt 5; $i++) {
            try {
                [System.Windows.Forms.Clipboard]::SetDataObject($data, $true)
                return $true
            } catch {
                Start-Sleep -Milliseconds 250
            }
        }
    } catch {}
    return $false
}

function escapeRegExp(v){return v.replace(/[.*+?^`$`{}()|[\]\\]/g,'\\$&');}
function setActiveMatch(preId,idx,behavior){
  var pre=document.getElementById(preId);
  if(!pre){return;}
  var matches=pre.querySelectorAll('mark.match');
  matches.forEach(function(m){m.classList.remove('match-active');});
  if(!matches.length){
    pre.dataset.matchIndex='-1';
    return;
  }
  if(idx>=matches.length){idx=0;}
  if(idx<0){idx=matches.length-1;}
  pre.dataset.matchIndex=idx;
  matches[idx].classList.add('match-active');
  matches[idx].scrollIntoView({block:'center',behavior:behavior||'smooth'});
}
function highlightLog(preId,inputId){
  var pre=document.getElementById(preId);
  var input=document.getElementById(inputId);
  if(!pre.dataset.raw){pre.dataset.raw=pre.textContent;}
  var raw=pre.dataset.raw;
  var q=input.value.trim();
  pre.dataset.lastQuery=q;
  pre.dataset.matchIndex='-1';
  if(!q){
    pre.innerHTML=raw.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
    return;
  }
  var esc=escapeRegExp(q);
  var html=raw.replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;');
  pre.innerHTML=html.replace(new RegExp(esc,'gi'),function(m){return '<mark class=\"match\">'+m+'</mark>';});
  var matches=pre.querySelectorAll('mark.match');
  if(matches.length){setActiveMatch(preId,0,'smooth');}
}
function searchOrNextLog(preId,inputId){
  var pre=document.getElementById(preId);
  var input=document.getElementById(inputId);
  if(!pre||!input){return;}
  var q=input.value.trim();
  var matches=pre.querySelectorAll('mark.match');
  if(q && pre.dataset.lastQuery===q && matches.length){
    jumpMatch(preId,1);
    return;
  }
  highlightLog(preId,inputId);
}
function jumpMatch(preId,dir){
  var pre=document.getElementById(preId);
  var matches=pre.querySelectorAll('mark.match');
  if(!matches.length){return;}
  var idx=parseInt(pre.dataset.matchIndex||'-1',10);
  idx=idx+dir;
  setActiveMatch(preId,idx,'smooth');
}
function handleSearchKey(evt,preId){
  if(evt.key!=='Enter'){return;}
  evt.preventDefault();
  var inputId=preId==='events'?'eventsSearch':'exqlogsSearch'+preId.replace('exqlogs','');
  if(document.getElementById(inputId)){searchOrNextLog(preId,inputId);}
  if(evt.shiftKey){jumpMatch(preId,-1);}
}

function Invoke-TestJobDiagnosis {
    $Script:CollectionErrors = New-Object System.Collections.Generic.List[object]
    if ($Script:LaunchSource -eq "github") {
        Set-TestJobProgress 2 "Online versie geladen van publieke bron"
    } else {
        Set-TestJobProgress 2 "Lokale fallbackversie gestart"
    }
    Set-TestJobProgress 5 "Test-Job diagnose starten"

    Set-TestJobProgress 12 "Basisgegevens parallel verzamelen"
    $parallelResults = Invoke-ParallelTaskScripts -MaxWorkers 7 -Tasks @(
        [pscustomobject]@{
            Name = "installed_apps"
            ScriptText = @"
$(Get-FunctionSource 'Get-InstalledApps')
Get-InstalledApps
"@
            Arguments = @()
            Default = @()
        },
        [pscustomobject]@{
            Name = "windows_kb_updates"
            ScriptText = @"
$(Get-FunctionSource 'Get-WindowsKbUpdates')
Get-WindowsKbUpdates
"@
            Arguments = @()
            Default = @()
        },
        [pscustomobject]@{
            Name = "event_logs"
            ScriptText = @"
$(Get-FunctionSource 'Get-EventLogRows')
Get-EventLogRows
"@
            Arguments = @()
            Default = @()
        },
        [pscustomobject]@{
            Name = "exquise_logs"
            ScriptText = @"
$(Get-FunctionSource 'Get-ExquiseLogs')
Get-ExquiseLogs
"@
            Arguments = @()
            Default = @()
        },
        [pscustomobject]@{
            Name = "wan"
            ScriptText = @"
$(Get-FunctionSource 'Get-WanInfo')
Get-WanInfo
"@
            Arguments = @()
            Default = @([pscustomobject]@{ Ip = "Onbekend"; Isp = "Onbekend"; As = "" })
        }
    )

    Set-TestJobProgress 30 "Geinstalleerde software indelen"
    $apps = @($parallelResults["installed_apps"])
    $splitApps = Invoke-SafeCheck "split_apps" { Split-InstalledApps $apps } ([pscustomobject]@{ Software = @(); Updates = @() })

    Set-TestJobProgress 42 "Firewall endpoints testen"
    $domainTests = Invoke-SafeCheck "domain_tests" { Test-DomainSet } ([pscustomobject]@{ Extensive = @(); Basic = @(); Failures = @() })

    Set-TestJobProgress 58 "LAN apparaten scannen"
    $lanInfo = Invoke-SafeCheck "lan_scan" { Get-LanScan } ([pscustomobject]@{ Subnet = ""; Total = 0; SonosCount = 0; SonosFound = $false; YealinkCount = 0; YealinkFound = $false; Devices = @() })

    Set-TestJobProgress 70 "Systeem, beveiliging en netwerkdetails verzamelen"
    $data = [pscustomobject]@{
        Timestamp = Get-Date
        Naam = ""
        Praktijk = ""
        Ticketnummer = ""
        Licentie = ""
        ComputerName = Invoke-SafeCheck "computer_name" { $env:COMPUTERNAME } ""
        LoggedUser = Invoke-SafeCheck "logged_user" { "$env:USERDOMAIN\$env:USERNAME" } ""
        Os = Invoke-SafeCheck "os_info" { Get-OsInfo } ([pscustomobject]@{ Full = "Niet beschikbaar"; Build = ""; Version = "" })
        SystemModel = Invoke-SafeCheck "system_model" { Get-SystemModel } ""
        Cpu = Invoke-SafeCheck "cpu_info" { Get-CpuInfo } ([pscustomobject]@{ Name = "Niet beschikbaar"; Cores = 0; Logical = 0; LaunchYear = $null })
        Memory = Invoke-SafeCheck "memory_info" { Get-MemoryInfo } ([pscustomobject]@{ TotalGb = "?"; AvailableGb = "?"; Percent = "?" })
        Disks = @(Invoke-SafeCheck "disks" { Get-DiskInfo } @())
        Uptime = Invoke-SafeCheck "uptime" { Get-UptimeInfo } ([pscustomobject]@{ Text = "-" })
        FastBoot = Invoke-SafeCheck "fastboot" { Get-FastBoot } $null
        RebootPending = Invoke-SafeCheck "reboot_pending" { Get-RebootPending } $null
        ProxyEnabled = Invoke-SafeCheck "proxy" { Get-ProxyEnabled } $null
        WebView2 = Invoke-SafeCheck "webview2" { Get-WebView2Info } ([pscustomobject]@{ Installed = $false; Version = "" })
        DotNet = Invoke-SafeCheck "dotnet" { Get-DotNetVersion } ""
        LastWindowsUpdate = Invoke-SafeCheck "last_windows_update" { Get-LastWindowsUpdate } ""
        WindowsInstall = Invoke-SafeCheck "windows_install_date" { Get-WindowsInstallDate $apps } ([pscustomobject]@{ Date = ""; Source = "onbekend" })
        KbUpdates = @($parallelResults["windows_kb_updates"])
        Antivirus = @(Invoke-SafeCheck "antivirus" { Get-AntivirusProducts } @())
        SmartAppControl = Invoke-SafeCheck "smart_app_control" { Get-SmartAppControl } ([pscustomobject]@{ State = "Niet beschikbaar" })
        ExquiseProcesses = @(Invoke-SafeCheck "exquise_processes" { Get-ProcessInfo "exquise" } @())
        IasProcesses = @(Invoke-SafeCheck "ias_processes" { Get-ProcessInfo "imageacquisition" } @())
        Adapters = @(Invoke-SafeCheck "network_adapters" { Get-NetworkAdapters } @())
        Gateway = Invoke-SafeCheck "gateway" { Get-GatewayIp } ""
        DnsServers = Invoke-SafeCheck "dns_servers" { Get-DnsServers } ""
        Wan = if ($parallelResults["wan"] -and $parallelResults["wan"].Count -gt 0) { $parallelResults["wan"][0] } else { [pscustomobject]@{ Ip = "Onbekend"; Isp = "Onbekend"; As = "" } }
        Network = $domainTests
        Lan = $lanInfo
        EventLogs = @($parallelResults["event_logs"])
        ExquiseLogs = @($parallelResults["exquise_logs"])
        AppChecks = @(Invoke-SafeCheck "app_checks" { Get-AppChecks $splitApps.Software } @())
        Software = @($splitApps.Software)
        Updates = @($splitApps.Updates)
        CollectionErrors = $Script:CollectionErrors
        ClipboardOk = $false
        ReportFileName = ""
    }
    Set-TestJobProgress 82 "Diagnosegegevens verzameld"
    return $data
}

            if ($null -eq $oldWorker) { Remove-Item Env:\TESTJOB_WORKER -ErrorAction SilentlyContinue } else { $env:TESTJOB_WORKER = $oldWorker }
            if ($null -eq $oldResult) { Remove-Item Env:\TESTJOB_RESULT_PATH -ErrorAction SilentlyContinue } else { $env:TESTJOB_RESULT_PATH = $oldResult }
            if ($null -eq $oldProgress) { Remove-Item Env:\TESTJOB_PROGRESS_PATH -ErrorAction SilentlyContinue } else { $env:TESTJOB_PROGRESS_PATH = $oldProgress }
            if ($null -eq $oldOutput) { Remove-Item Env:\TESTJOB_OUTPUT_DIR -ErrorAction SilentlyContinue } else { $env:TESTJOB_OUTPUT_DIR = $oldOutput }
        }
        $pollTimer.Start()
    })

    $form.Add_FormClosed({
        try { if ($pollTimer) { $pollTimer.Stop(); $pollTimer.Dispose() } } catch { }
        try { if ($Script:WorkerProcess -and -not $Script:WorkerProcess.HasExited) { $Script:WorkerProcess.Kill() } } catch { }
        Remove-Item -LiteralPath $Script:WorkerResultPath,$Script:WorkerProgressPath -Force -ErrorAction SilentlyContinue
    })

    [System.Windows.Forms.Application]::Run($form)
}
